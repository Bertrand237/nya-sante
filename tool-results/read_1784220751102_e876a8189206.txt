   133→  return (
   134→    <Card className="group hover:shadow-md transition-shadow">
   135→      <CardHeader className="pb-3">
   136→        <div className="flex items-start justify-between">

   139→              <Icon className="w-4.5 h-4.5 text-primary" />
   140→            </div>
   141→            <CardTitle className="text-base">{title}</CardTitle>
   142→          </div>
   143→          {status && (
   144→            <Badge variant="outline" className={`text-xs ${statusConfig[status].color} border-0`}>
   145→              {statusConfig[status].label}
   146→            </Badge>
   147→          )}
   148→        </div>
   149→      </CardHeader>

   156→            ))}

   158→        )}

   160→    </Card>
